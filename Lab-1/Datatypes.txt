package samplejava;

public class PrimitiveDatatypes {

	public static void main(String[] args) {
		// This is variable Declaration
		byte age;  //It contains 8bits
		short strength; //It contains 16bits
		int standard; // It contains 32bits
		long RollNo; // It contains 64bits
		float  marks; // It contains 32bits
		double Topscore; // It contains 64bits
		boolean pass; //boolean store (true or false)
		char gender ; // It contains 16bits
		//This is variable initialisation
		age = 23;
		strength = 40;
		standard = 7;
		RollNo= 1373256126l;
		marks =25.5f;
		Topscore=99.5d;
		pass=false;
		gender ='F';
		  System.out.println("Student age:-"+age);  //It prints 23
		  System.out.println("Student strength :-"+strength); // It prints 40
		  System.out.println("Student standard :-"+ standard); // It prints 7
		  System.out.println("Student RollNo :-"+RollNo); // It prints 1373256126l
		  System.out.println("Student marks:-"+marks); // It prints25.5
		  System.out.println("Student Topscore:-"+Topscore); // It prints 99.5
		  System.out.println("Student pass:-"+pass); //It prints false
		  System.out.println("Student gender:-"+gender); //It prints F
	}

}
